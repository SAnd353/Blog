    if len(list) == 0:
        print("no one likes this")
    elif len(list) == 1:
        print(str(list[0]) + " likes this")
    elif len(list) >= 4:
        print(list[0] + ", " + list[1] + " and " + str(len(list) -2) + " others like this")
    else:
        print(", ".join(list[0:-1]) + " and " + str(list[-1]) + " like this")

Like = False
Dislike = False

def like_or_dislike(list):
    global Like
    global Dislike
    for i in range(len(list)):
        if list[i] == "Like":
            Dislike = False
            Like = not(Like)
        if list[i] == "Dislike":
            Like = False
            Dislike = not(Dislike)
    if Like == True:
        print("Like")
    elif Dislike == True:
        print("Dislike")
    else:
        print("Nothing")
    
like_or_dislike(["Dislike"])
like_or_dislike(["Like", "Like"])
like_or_dislike(["Dislike", "Like"])
